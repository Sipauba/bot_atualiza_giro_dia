
// dias uteis de venda
SELECT 60+count(diavendas) FROM PCDIASUTEIS 
WHERE DIAVENDAS='S' 
AND CODFILIAL=6 
AND DATA BETWEEN to_date('01/'||to_char(SYSDATE,'mm/yyyy'))
AND TRUNC(SYSDATE)-1




-- MUDAR QUANTIDADE DIAS UTEIS
UPDATE PCEST 
SET QTGIRODIA=round((QTVENDMES+QTVENDMES1+QTVENDMES2+QTVENDMES3)/74,2)
WHERE CODFILIAL IN (3,4,5,6,7,70,14,17,18,19,20,61);
COMMIT;
